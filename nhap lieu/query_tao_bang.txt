CREATE TABLE tb_plan
(
  	id_plan BIGINT NOT NULL AUTO_INCREMENT,
  
	SQL_oid_line TEXT, -- trỏ tới [HiPro].[dbo].[pro_chuyen].oid_mapping
  	line TEXT, -- [HiPro].[dbo].[pro_chuyen].stt
  	SQL_oid_ma_hang TEXT, -- trỏ tới [HiPro].[dbo].[DM_SanPham].oid
  	style TEXT, -- [HiPro].[dbo].[DM_SanPham].MaSanPham
	
	plan_date DATETIME,
	actual_date DATETIME,
	A3_documentation TEXT,
    percent_rate INT,

	created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  	updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
	updated_by TEXT,

	PRIMARY KEY (id_plan)
);

CREATE TABLE tb_co
(
	id_co BIGINT NOT NULL AUTO_INCREMENT,
	id_plan BIGINT,
	
	CO_begin_date DATETIME, -- ngày bắt đầu CO
	CO_end_date DATETIME, -- ngày kết thúc CO
	first_garment_of_new_style DATETIME, -- thời gian hoàn thành chuyển đổi
	
	carry_over INT CHECK (carry_over IN (0,1)), -- lặp lại
	buyer TEXT, -- khách hàng
	production_style TEXT, -- loại
	
	SAM INT,
	staff TEXT,
	
	quota DECIMAL(15,2), -- định mức
	eff_1 DECIMAL(15,2), -- eff ngày 1
	
	target_of_COPT DECIMAL(15,2), -- Mục Tiêu COPT
	COPT DECIMAL(15,2), -- Thời Gian Chuyển Đổi Trong
	target_of_COT DECIMAL(15,2), -- Mục Tiêu COT
	COT DECIMAL(15,2), -- Thời Gian Chuyển Đổi
	
	created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  	updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
	updated_by TEXT,
	
    PRIMARY KEY (id_co),
    UNIQUE (id_plan, id_co)
);
